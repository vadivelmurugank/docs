SSH SERVER SETUP¶

sed -i -e “s/PermitRootLogin [Nn]o/PermitRootLogin yes/” /etc/ssh/sshd_config sed -i -e “s/[#]*ClientAliveInterval.*[0-9]*/ClientAliveInterval 10/” /etc/ssh/sshd_config sed -i -e “s/[#]*ClientAliveCountMax.*[0-9]*/ClientAliveCountMax 6/” /etc/ssh/sshd_config sed -i -e “s/[#]*StrictModes.*[a-z]*/StrictModes no/” /etc/ssh/sshd_config killall sshd

rm -rf /root/.ssh mkdir -p /root/.ssh ssh-keygen -v -N “” -f /root/.ssh/id_rsa touch /root/.ssh/authorized_keys chmod 640 /root/.ssh/authorized_keys chmod 700 /root/.ssh chmod 600 /root/.ssh/id_rsa chmod 640 /root/.ssh/id_rsa.pub

SERVER KEYGEN¶

# Remove Existing key entry for the linecard ssh-keygen -R 127.3.12.1

# Verify ssh connection to the linecard
ssh -v -o PreferredAuthentications=password -o ServerAliveInterval=10 -o ServerAliveCountMax=20 127.3.12.1 cat /release.inf
# ssh to linecard
ssh -v -o ServerAliveInterval=10 -o ServerAliveCountMax=20 127.3.12.1
