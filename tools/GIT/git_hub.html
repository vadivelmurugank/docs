GIT SETUP¶

echo “Hello I am Git” >> README.md git init git add README.md git commit -m “first commit”

# Add to remote repo git remote add origin https://github.com/vadivelmurugank/hello.git

GIT gh-pages branch¶

#create orphaned gh-pages branch
git checkout –orphan gh-pages git branch
# remove all cached files to be committed.
git rm -rf .

echo “Hello Pages!!” > index.html git add index.html git commit -m ‘initial gh-pages commit’

# git push to gh-pages branch
git push origin gh-pages
# add gh-pages as submodule
git checkout master git branch

# Create folder “publish” and add gh-pages as submodule. # Changing the folder to “publish” will set to gh-pages branch git submodule add -b gh-pages https://github.com/vadivelmurugank/<< GIT >>.git publish git submodule init

git commit -m “added gh-pages as submodule” git push origin master

# Push changes into gh-pages cd publish git add <<file>> git commit -m “file updates” git push -u origin gh-pages

cd .. git add publish git commit -m “update publish” git push -u origin master
