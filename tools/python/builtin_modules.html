id():

id of an object, which is object address in memory

a=[1, 2, 3]
hex(id(a))
    '0x6ffffad1908'

# Get Ctype Pointer address for list (a)
import ctypes, sys
ptr=ctypes.cast(hex(id(a)), ctypes.POINTER(ctypes.c_uint8))
sys.getsizeof(ptr)
    144

bytes(ptr[i] for i in range[144])

range():

range(end)
range(start,end)
range(start,end,increment)

sorted():

sorted(range(1,10))
 [1, 2, 3, 4, 5, 6, 7, 8, 9]
 sorted(zip(range(1,10), range(9,12)))
 [(1, 9), (2, 10), (3, 11)]

lambda anonymous function:

lambda (arg1, arg2, ...): << body of function>>

lambda x: (x * 2)
lambda x,y,z: (x + y + z)

zip():

zip of two lists, tuples,..
a = [1 , 2, 3]
b = [5, 6, 8]
list(zip(a,b))
[(1, 5), (2, 6), (3, 8)]

# Zip of Dictionaries keys and values
a={1: 'a', 2: 'b'}
b={3: 'c', 4: 'd'}
list(zip(a,b))
    [(1, 3), (2, 4)]

list(zip(a.values(),b.values()))
    [('a', 'c'), ('b', 'd')]

enumerate():

enumerate ( sequences, start=0)
b = [5, 6, 8]
list(enumerate(b,start=1))
    [(1, 5), (2, 6), (3, 8)]
filter():

filter( function, iterable)

# Check if number is divisible by 3
list(filter(lambda x:  (x%3 == 0), range(90,99)))
[90, 93, 96]

map():

map( function(1arg), iterable, ..)
map( function(1arg, 2arg), iterable1,iterable2 ..)

map(lambda x: (x * 2), range(x,10))
    ==> equivalent to [x * 2 for x in range(1,10)]
map(lambda x,y,z: (x + y + z), range(1,10), range(10,20), range(20,30))
    ==> [x+y+z for x, y, z in zip(range(1,10),range(10,20),range(20,30))]

compile():

compile(code, filename / '<string>', eval or exec)

mode :  exec - source  contains sequence of statements
                - "list(map(lambda a: a ^ a, range(1,10)))"
        eval - source contains single expressions
                "a + b"
